Sub RunPythonScript()
    Dim objShell As Object
    Dim PythonExePath As String
    Dim PythonScriptPath As String
    Dim Command As String
    
    ' Path to the python.exe
    PythonExePath = """C:\Users\USUARIO\anaconda3\envs\HR\python.exe"""
    
    ' Path to the Python script
    PythonScriptPath = """C:\Users\USUARIO\Desktop\studious-winner\run.py"""
    
    ' Construct the command to run the Python script
    Command = "cmd /k " & PythonExePath & " " & PythonScriptPath
    
    ' Create a shell object
    Set objShell = CreateObject("WScript.Shell")
    
    ' Check if the shell object is created successfully
    If Not objShell Is Nothing Then
        ' Construct the command to run the Python script
        Command = "cmd /k """ & PythonExePath & """ """ & PythonScriptPath & """"
        
        ' Run the command
        objShell.Run Command
    Else
        MsgBox "Failed to create WScript.Shell object"
    End If
End Sub


Sub RunPythonScript()
    Dim objShell As Object
    Dim BatchFilePath As String
    
    ' Path to the batch file
    BatchFilePath = "C:\Users\USUARIO\Desktop\studious-winner\run_py.bat"
    
    ' Full path to the Excel file
    ExcelFilePath = ThisWorkbook.FullName
    
    ' Extract the directory path from the Excel file path
    DirectoryPath = Left(ExcelFilePath, InStrRev(ExcelFilePath, "\"))
    
    ' Change the current working directory to the directory containing the Excel file
    ChDir DirectoryPath
    
    ' Create a shell object
    Set objShell = CreateObject("WScript.Shell")
    
    ' Check if the shell object is created successfully
    If Not objShell Is Nothing Then
        ' Run the batch file
        objShell.Run """" & BatchFilePath & """"
    Else
        MsgBox "Failed to create WScript.Shell object"
    End If
    
    
End Sub
